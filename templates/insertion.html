How Insertion Sort Works

Insertion sort builds a sorted array one element at a time by taking each element and inserting it into its correct position within the already sorted portion like organizing cards in your hand

• Pick an element from the unsorted portion (key)

• Compare it with elements in the sorted portion

• Shift larger elements right and insert the key

def insertion_sort(arr):
    for i in range(1, len(arr)):
        key = arr[i]
        j = i - 1
        while j >= 0 and arr[j] > key:
            arr[j + 1] = arr[j]
            j -= 1
        arr[j + 1] = key
    return arr

fn insertion_sort(arr: &mut [i32]) {
    for i in 1..arr.len() {
        let key = arr[i];
        let mut j = i;
        while j > 0 && arr[j - 1] > key {
            arr[j] = arr[j - 1];
            j -= 1;
        }
        arr[j] = key;
    }
}
fn main() {
    let mut arr = [5, 2, 8, 1, 9];
    insertion_sort(&mut arr);
    println!("{:?}", arr);
}
